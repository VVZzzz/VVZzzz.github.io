using namespace std;

using iter = vector<int>::iterator;

iter f1(iter beg, iter end, int object)
{
    iter ptr = beg;
    while (ptr != end)
    {
        if (*ptr++ == object)
            return --ptr;
    }
    return ptr;
}

int main()
{
    vector<int> test;
    int count=10;
    while (count != 0)
    {
        test.push_back(count--);
    }

    auto i = test.begin();

    **//返回目标的下标**
    cout << f1(i, i + 8, 5)-i << endl;


    system("pause");
    return 0; 
}
